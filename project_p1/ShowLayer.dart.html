import 'package:flutter/material.dart';
import 'dart:ui';
class GalleryPage extends StatefulWidget {
  @override
  _GalleryPageState createState() => _GalleryPageState();
}

List<String> images = [
  'images/testLayout.jpg',
  'images/testLayout.jpg',
];

class _GalleryPageState extends State<GalleryPage> {

  final _key = GlobalKey();
  String _currentImage = images[0];
  String _currentFloor = "Floor " + 1.toString();


  double x_ratio = 0.5;
  double y_ratio = 0.5;
  final TransformationController _controller = TransformationController();

  @override
  Widget build(BuildContext context) {
    double width = MediaQuery.of(context).size.width - 20;
    double height = MediaQuery.of(context).size.height - 60;
    return Scaffold(
      body: SafeArea(
        child: Container(
          padding: EdgeInsets.all(16),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              Text(
                'SB123',
                style: TextStyle(
                    fontSize: 30,
                    fontFamily: 'avenir',
                    fontWeight: FontWeight.bold),
              ),
              Text(
                _currentFloor,
                style: TextStyle(fontSize: 18, fontFamily: 'avenir'),
              ),
              Expanded(
                child: InteractiveViewer(
                  boundaryMargin: EdgeInsets.all(10),
                  transformationController: _controller,
                  maxScale: 5.0,
                  child: Container(
                    decoration: new BoxDecoration(color: Colors.white),
                    height: height,
                    width: width,
                    child: Stack(
                      children: <Widget>[
                        Image.asset(_currentImage, key: _key,),
                        Positioned(
                          bottom: height * y_ratio, right: width * x_ratio, //give the values according to your requirement
                          child: Icon(Icons.location_on),
                        ),
                      ],
                    ),
                  ),
                ),
              ),
              Container(
                height: 80,
                child: ListView.builder(
                    scrollDirection: Axis.horizontal,
                    itemCount: images.length,
                    itemBuilder: (context, index) {
                      return InkWell(
                        onTap: () {
                          setState(() {
                            _currentImage = images[index];
                            _currentFloor = "Floor " + (index + 1).toString();
                            _controller.value = Matrix4.identity();

                          });
                        },
                        child: Container(
                          margin: const EdgeInsets.all(8),
                          padding: const EdgeInsets.all(2),
                          color: Colors.black12,
                          child: Image.asset(images[index]),
                        ),
                      );
                    }),
              ),
            ],
          ),
        ),
      ),
    );
  }
}
